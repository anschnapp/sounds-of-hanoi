define :qsort do |list|
  a = list.shift
  
  
  low = []
  high = []
  list.each do |element|
    play element, release: 0.1
    sleep 0.03
    if element <= a
      low << element
    else
      high << element
    end
  end
  if low.empty?
    sortedLow = []
  else
    
    sortedLow = qsort low
  end
  if high.empty?
    sortedHigh = []
  else
    sortedHigh = qsort high
  end
  result = []
  result = sortedLow
  result << a
  result = result + sortedHigh
  return result
end
